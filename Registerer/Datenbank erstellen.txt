CREATE DATABASE students;
USE students;
CREATE TABLE student (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, firstname VARCHAR(30) NOT NULL, lastname VARCHAR(30) NOT NULL, contact INT(6) NOT NULL);
CREATE TABLE contact ( id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, firstname VARCHAR(30), lastname VARCHAR(30), phone VARCHAR(30), mobile VARCHAR(30), email VARCHAR(50));
CREATE TABLE course (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,course VARCHAR(30) NOT NULL, teacher INT(6) NOT NULL);
CREATE TABLE teacher (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, fistname VARCHAR(30) NOT NULL, lastname VARCHAR(30) NOT NULL, phone VARCHAR(30), mobile VARCHAR(30), email VARCHAR(50));
CREATE TABLE attendance ( id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, student INT(6) NOT NULL, course INT(6) NOT NULL, date DATE NOT NUll);
